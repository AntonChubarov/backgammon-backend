func RollOneDice() int {
        return randomInt(1, 6)
}

func randomInt(min, max int) int {
        nBig, err := rand.Int(rand.Reader, big.NewInt(int64(max - min + 1)))
        if err != nil {
                panic(err)
        }
        n := nBig.Int64()
        return min + int(n)
}
